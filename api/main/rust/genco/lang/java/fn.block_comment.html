Expand description

Format a block comment, starting with /**, and ending in */.

Examples

use genco::prelude::*;
use std::iter;

let toks = quote! {
    #(java::block_comment(vec!["first line", "second line"]))
    #(java::block_comment(iter::empty::<&str>()))
    #(java::block_comment(vec!["third line"]))
};

assert_eq!(
    vec![
        "/**",
        " * first line",
        " * second line",
        " */",
        "/**",
        " * third line",
        " */",
    ],
    toks.to_file_vec()?
);
